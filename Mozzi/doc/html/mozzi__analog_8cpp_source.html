00212                                 pin -= 14; // allow for channel or pin numbers

00214 
00215 #if defined(__AVR_ATmega32U4__)
00216 
00217                 pin = analogPinToChannel(pin);
00218                 ADCSRB = (ADCSRB & ~(1 << MUX5)) | (((pin >> 3) & 0x01) << MUX5);
00219 #elif defined(ADCSRB) && defined(MUX5)
00220                 // the MUX5 bit of ADCSRB selects whether we're reading from channels
00221                 // 0 to 7 (MUX5 low) or 8 to 15 (MUX5 high).
00222                 ADCSRB = (ADCSRB & ~(1 << MUX5)) | (((pin >> 3) & 0x01) << MUX5);
00223 #endif
00224 
00225                 // set the analog reference (high two bits of ADMUX) and select the
00226                 // channel (low 4 bits).  this also sets ADLAR (left-adjust result)
00227                 // to 0 (the default).
00228 #if defined(ADMUX)
00229 
00230                 ADMUX = (analog_reference << 6) | (pin & 0x07);
00231 #endif
00232 
00233                 // without a delay, we seem to read from the wrong channel
00234                 //delay(1);

00237                 // start the conversion
00238                 sbi(ADCSRA, ADSC);

00253 int receiveAnalogRead()
00254 {
00255                 unsigned char low, high;
00256 #if defined(ADCSRA) && defined(ADCL)
00257                 // ADSC is cleared when the conversion finishes
00258                 while (bit_is_set(ADCSRA, ADSC))
00259                                 ;
00260 
00261                 // we have to read ADCL first; doing so locks both ADCL
00262                 // and ADCH until ADCH is read.  reading ADCL second would
00263                 // cause the results of each conversion to be discarded,
00264                 // as ADCL and ADCH would be locked when it completed.
00265                 low  = ADCL;
00266                 high = ADCH;
00267 #else
00268                 // we dont have an ADC, return 0
00269                 low  = 0;
00270                 high = 0;
00271 #endif
00272 
00273                 // combine the two bytes
00274                 return (high << 8) | low;
00275 }

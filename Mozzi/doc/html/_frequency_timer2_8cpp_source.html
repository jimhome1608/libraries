00067 void FrequencyTimer2::setPeriod(unsigned long period)
00068 {
00069     uint8_t pre, top;
00070   
00071     if ( period == 0) period = 1;
00072     period *= clockCyclesPerMicrosecond();
00073  
00074     period /= 2;            // we work with half-cycles before the toggle 
00075     if ( period <= 256) {
00076                 pre = 1;
00077                 top = period-1;
00078     } else if ( period <= 256L*8) {
00079                 pre = 2;
00080                 top = period/8-1;
00081     } else if ( period <= 256L*32) {
00082                 pre = 3;
00083                 top = period/32-1;
00084     } else if ( period <= 256L*64) {
00085                 pre = 4;
00086                 top = period/64-1;
00087     } else if ( period <= 256L*128) {
00088                 pre = 5;
00089                 top = period/128-1;
00090     } else if ( period <= 256L*256) {
00091                 pre = 6;
00092                 top = period/256-1;
00093     } else if ( period <= 256L*1024) {
00094                 pre = 7;
00095                 top = period/1024-1;
00096     } else {
00097                 pre = 7;
00098                 top = 255;
00099     }
00100 
00101 #if defined(TCCR2A)
00102     TCCR2B = 0;
00103     TCCR2A = 0;
00104     TCNT2 = 0;
00105 #if defined(ASSR) && defined(AS2)
00106     ASSR &= ~_BV(AS2);    // use clock, not T2 pin
00107 #endif
00108     OCR2A = top;
00109     TCCR2A = (_BV(WGM21) | ( FrequencyTimer2::enabled ? _BV(COM2A0) : 0));
00110     TCCR2B = pre;
00111 #elif defined(TCCR2)
00112     TCCR2 = 0;
00113     TCNT2 = 0;
00114     ASSR &= ~_BV(AS2);    // use clock, not T2 pin
00115     OCR2 = top;
00116     TCCR2 = (_BV(WGM21) | ( FrequencyTimer2::enabled ? _BV(COM20) : 0)  | pre);
00117 #endif
00118 }

00120 unsigned long  FrequencyTimer2::getPeriod()
00121 {
00122 #if defined(TCCR2B)
00123     uint8_t p = (TCCR2B & 7);
00124     unsigned long v = OCR2A;
00125 #elif defined(TCCR2)
00126     uint8_t p = (TCCR2 & 7);
00127     unsigned long v = OCR2;
00128 #endif
00129     uint8_t shift;
00130   
00131     switch(p) {
00132       case 0 ... 1:
00133                 shift = 0;

00135       case 2:
00136                 shift = 3;
00137                 break;
00138       case 3:
00139                 shift = 5;
00140                 break;
00141       case 4:
00142                 shift = 6;
00143                 break;
00144       case 5:
00145                 shift = 7;
00146                 break;
00147       case 6:
00148                 shift = 8;
00149                 break;
00150       case 7:
00151                 shift = 10;
00152                 break;
00153     }
00154     return (((v+1) << (shift+1)) + 1) / clockCyclesPerMicrosecond();   // shift+1 converts from half-period to period
